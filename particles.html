Частицы (Партиклы)

Базовые фигуры

# Спавн частиц в точке
particleSpawn("minecraft:flame", x, y, z, 10, 0.5, 0.5, 0.5, 0.1)

# Линия из частиц
particleLine("minecraft:soul", 0, 64, 0, 10, 64, 10, 20, 0, 0, 0, 0)

# Круг из частиц
particleCircle("minecraft:soul_fire_flame", x, y, z, 3.0, 40, 0, 0, 0, 0)

# Спираль частиц
particleSpiral("minecraft:end_rod", x, y, z, 2.0, 5.0, 30, 0.1, 0.2, 0.1, 0.05)

Параметры dx, dy, dz задают разброс (spread), параметр speed — скорость разлета. Если speed > 0, то dx, dy, dz работают как направление.

Постоянные эффекты

async "magic_circle" {
    while (true) {
        particleCircle("minecraft:soul", x, y, z, 2.0, 20, 0, 0, 0, 0)
        await time(1)
    }
}
stop_task("magic_circle")

Привязка к костям NPC

# Непрерывный спавн частиц из кости NPC (на клиенте)
particleStartBone("fire_hands", knight, "right_hand", "minecraft:flame", 5, 0.1, 0.1, 0.1, 0.05)

# Остановка
particleStopBone("fire_hands")

Кости NPC: "head", "right_hand", "left_hand", "body" и другие в зависимости от модели.

Пример: магический круг

async "magic_circle" {
    val x = 0
    val y = 64
    val z = 0
    
    while (true) {
        particleCircle("minecraft:soul_fire_flame", x, y, z, 3.0, 40, 0, 0, 0, 0)
        particleSpiral("minecraft:end_rod", x, y, z, 2.0, 5.0, 30, 0.1, 0.2, 0.1, 0.05)
        await time(2)
    }
}

on position(player, 0, 64, 0, 3) -> in_circle {
    particleSpawn("minecraft:totem_of_undying", player.x, player.y + 1, player.z, 20, 0.5, 0.5, 0.5, 0.1)
    chat("Ты вошел в магический круг!")
}
