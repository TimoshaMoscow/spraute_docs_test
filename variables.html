Переменные и область видимости

Типы переменных

ТипОписаниеПример
valЛокальная переменнаяval x = 10
global valГлобальная в рамках скриптаglobal counter = 0
world valСохраняется между перезапускамиworld server_started = true
player.dataСессионные данные игрокаplayer.data["in_dialogue"] = true
player.savedDataПостоянные данные игрокаplayer.savedData["deaths"] = 0

Примеры использования

# Локальная переменная
val local_var = 10

# Глобальная для скрипта
global global_var = "доступна везде"

# Сохраняется между перезапусками сервера
world saved_var = 42

# Сессионные данные игрока (до перезапуска сервера)
player.data["in_dialogue"] = true

# Постоянные данные игрока (сохраняются в мире навсегда)
val p = getPlayer("Steve")
if (p.savedData["deaths"] == null) {
    p.savedData["deaths"] = 0
}
p.savedData["deaths"] = p.savedData["deaths"] + 1

Игрок:

player.name
player.hp
player.x, player.y, player.z
player.max_hp
player.pitch, player.yaw
player.look_x, player.look_y, player.look_z  # вектор взгляда
player.uuid

NPC:

npc.name
npc.show_name
npc.hp, npc.max_hp
npc.speed
npc.x, npc.y, npc.z
npc.yaw, npc.pitch
npc.model, npc.texture
npc.uuid

Присваивание свойств:

npc.hp = 100
npc.x = 10
npc.yaw = 90
