Комментарии

val x = 10  # Тоже комментарий

# Многострочные комментарии не поддерживаются,
# нужно использовать # на каждой строке

Функция range()

# range(n) — от 0 до n-1
for (i in range(5)) {
    chat("i = " + i)  # 0, 1, 2, 3, 4
}

# range(start, end) — от start до end-1
for (i in range(2, 7)) {
    chat("i = " + i)  # 2, 3, 4, 5, 6
}
