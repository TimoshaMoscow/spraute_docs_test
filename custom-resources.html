Кастомные ресурсы

Кастомные команды

create command mycmd {
    val p = _event_player
    val args = _event_args
    
    if (p != null) {
        if (size(args) > 0) {
            val firstArg = args[0]
            chat(p, "Команда вызвана с аргументом: " + firstArg)
        } else {
            chat(p, "Команда вызвана без аргументов")
        }
    }
}

Команда регистрируется в сервере и доступна всем игрокам. Аргументы автоматически конвертируются: числа в Double, true/false в Boolean, остальное в строки.

Рецепты крафта

Форменный крафт (shaped)

create craft my_sword {
    type = "shaped"
    pattern_1 = " I "
    pattern_2 = " I "
    pattern_3 = " S "
    key_I = "minecraft:iron_ingot"
    key_S = "minecraft:stick"
    result = "spraute_engine:magic_stick"
    count = 1
}

Бесформенный крафт (shapeless)

create craft my_shapeless {
    type = "shapeless"
    ingredients = ["minecraft:iron_ingot", "minecraft:stick"]
    result = "minecraft:iron_sword"
    count = 1
}

Кастомный контейнер (UI с инвентарём)

create ui my_workbench {
    type = "container"
    size = [176, 166]
    background = "#00000000"
    
    image("bg") {
        pos = [0, 0]
        size = [176, 166]
        texture = "minecraft:textures/gui/container/dispenser.png"
    }
    
    player_inventory {
        pos = [8, 84]
    }
    
    slot(0) { pos = [62, 20] }
    slot(1) { pos = [80, 20] }
    slot(2) { pos = [98, 20] }
    
    button("craft_btn", "Скрафтить") {
        pos = [62, 50]
        size = [54, 20]
        onClick {
            val slot0 = getBlockSlot(bx, by, bz, 0)
            val slot1 = getBlockSlot(bx, by, bz, 1)
            if (slot0 == "minecraft:iron_ingot" && slot1 == "minecraft:stick") {
                # Даём результат
                giveItem(_event_player, "minecraft:iron_sword", 1)
                # Очищаем слоты
                setBlockSlot(bx, by, bz, 0, "")
                setBlockSlot(bx, by, bz, 1, "")
            }
        }
    }
}
